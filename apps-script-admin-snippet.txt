function doGet() {
  return ContentService.createTextOutput('Guard Tour API is running.');
}

function doPost(e) {
  try {
    const body = JSON.parse((e && e.postData && e.postData.contents) || '{}');
    const action = body.action;
    let result;

    // Guard actions
    if (action === 'loginGuard') result = loginGuard(body.guardId);
    else if (action === 'guardBootstrap') result = guardBootstrap(body.guardId, body.date);
    else if (action === 'getShiftPlan') result = getShiftPlan(body.guardId, body.date);
    else if (action === 'submitCheckpoint') result = submitCheckpoint(body.payload || {});
    else if (action === 'submitIncident') result = submitIncident(body.payload || {});
    else if (action === 'closeShift') result = closeShift(body.shiftId);

    // Supervisor/Admin actions
    else if (action === 'supervisorLogin') result = supervisorLogin(body.supervisorId, body.password);
    else if (action === 'getSupervisorDailySummary') result = getSupervisorDailySummary(body.supervisorId, body.date);
    else if (action === 'getDashboardCharts') result = getDashboardCharts(body.supervisorId, body.endDate, body.days);
    else if (action === 'upsertSupervisor') result = upsertSupervisor(body.payload || {});
    else if (action === 'changeSupervisorPassword') result = changeSupervisorPassword(body.supervisorId, body.newPassword);
    else if (action === 'listGuards') result = listGuards(body.supervisorId);
    else if (action === 'upsertGuard') result = upsertGuard(body.payload || {});
    else if (action === 'deleteGuard') result = deleteGuard(body.guardId);
    else if (action === 'deleteSupervisor') result = deleteSupervisor(body.supervisorId);
    else if (action === 'listCheckpoints') result = listCheckpoints();
    else if (action === 'listCheckLogs') result = listCheckLogs(body.supervisorId, body.date, body.guardId, body.status);
    else if (action === 'listIncidents') result = listIncidents(body.supervisorId, body.date, body.guardId, body.status);
    else if (action === 'upsertCheckpoint') result = upsertCheckpoint(body.payload || {});
    else if (action === 'deleteCheckpoint') result = deleteCheckpoint(body.checkpointId);
    else if (action === 'listShifts') result = listShifts(body.date, body.guardId, body.supervisorId);
    else if (action === 'upsertShift') result = upsertShift(body.payload || {});
    else if (action === 'listShiftCheckpoints') result = listShiftCheckpoints(body.shiftId);
    else if (action === 'replaceShiftCheckpoints') result = replaceShiftCheckpoints(body.shiftId, body.items || []);
    else if (action === 'listShiftTemplates') result = listShiftTemplates(body.supervisorId);
    else if (action === 'upsertShiftTemplate') result = upsertShiftTemplate(body.payload || {});
    else if (action === 'deleteShiftTemplate') result = deleteShiftTemplate(body.templateId);
    else if (action === 'listTemplateCheckpoints') result = listTemplateCheckpoints(body.templateId);
    else if (action === 'replaceTemplateCheckpoints') result = replaceTemplateCheckpoints(body.templateId, body.items || []);
    else if (action === 'generateShiftsFromTemplates') result = generateShiftsFromTemplates(body.date, body.supervisorId);
    else if (action === 'testDriveWrite') result = testDriveWrite();

    else throw new Error('Unknown action');

    return jsonOut({ ok: true, data: result });
  } catch (err) {
    return jsonOut({ ok: false, error: String(err.message || err) });
  }
}

function jsonOut(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

/***** GUARD MAIN FUNCTIONS *****/
function loginGuard(guardId) {
  if (!guardId) throw new Error('guardId is required');
  const guards = getRows(SHEET.GUARDS);
  const g = guards.find(r =>
    String(r.guard_id) === String(guardId) &&
    String(r.status || '').toLowerCase() === 'active'
  );
  if (!g) throw new Error('Guard not found or inactive');
  return { guard_id: g.guard_id, name: g.name, supervisor_id: g.supervisor_id };
}

function guardBootstrap(guardId, dateStr) {
  if (!guardId) throw new Error('guardId is required');
  const targetDate = toDateKey(dateStr || fmtDateOnly(new Date()));
  const guard = loginGuard(guardId);
  const shifts = getShiftPlan(guardId, targetDate);
  const checkpoints = getRows(SHEET.CHECKPOINTS);
  const checkpointMap = toMap(checkpoints, 'checkpoint_id');
  const checkpointQrMap = {};

  checkpoints.forEach(cp => {
    const id = String(cp.checkpoint_id || '').trim();
    if (!id) return;
    checkpointQrMap[id] = String(cp.qr_text || id).trim();
  });

  const progressByShift = {};
  getRows(SHEET.CHECKLOGS)
    .filter(r =>
      String(r.guard_id || '') === String(guardId) &&
      toDateKey(r.scan_time) === targetDate
    )
    .forEach(r => {
      const status = String(r.status || '').toUpperCase();
      if (status !== 'ONTIME' && status !== 'LATE') return;
      const shiftId = String(r.shift_id || '').trim();
      const checkpointId = String(r.checkpoint_id || '').trim();
      if (!shiftId || !checkpointId) return;
      if (!progressByShift[shiftId]) progressByShift[shiftId] = {};
      progressByShift[shiftId][checkpointId] = Number(progressByShift[shiftId][checkpointId] || 0) + 1;
    });

  const enrichedShifts = shifts.map(shift => ({
    shift_id: shift.shift_id,
    date: shift.date,
    shift_name: shift.shift_name,
    start_time: shift.start_time,
    end_time: shift.end_time,
    rounds_required: shift.rounds_required,
    status: shift.status,
    checkpoints: (Array.isArray(shift.checkpoints) ? shift.checkpoints : []).map(point => ({
      seq_no: point.seq_no,
      round_no: point.round_no,
      checkpoint_id: point.checkpoint_id,
      checkpoint_name: point.checkpoint_name,
      planned_time: point.planned_time,
      sla_min: point.sla_min,
      qr_text: checkpointMap[String(point.checkpoint_id || '')]
        ? String(checkpointMap[String(point.checkpoint_id || '')].qr_text || point.checkpoint_id)
        : String(point.checkpoint_id || '')
    }))
  }));

  return {
    date: targetDate,
    guard: guard,
    shifts: enrichedShifts,
    checkpoint_qr_map: checkpointQrMap,
    progress_by_shift: progressByShift
  };
}

function getShiftPlan(guardId, dateStr) {
  if (!guardId || !dateStr) throw new Error('guardId and date are required');
  const targetDate = toDateKey(dateStr);

  ensureGuardShiftsFromTemplates(guardId, dateStr);

  const shifts = getRows(SHEET.SHIFTS).filter(r =>
    String(r.guard_id) === String(guardId) &&
    toDateKey(r.date) === targetDate
  );

  const scRows = getRows(SHEET.SHIFT_CHECKPOINTS);
  const cps = getRows(SHEET.CHECKPOINTS);
  const cpMap = toMap(cps, 'checkpoint_id');

  return shifts.map(s => {
    const points = scRows
      .filter(x => String(x.shift_id) === String(s.shift_id))
      .sort((a, b) =>
        Number(a.round_no || 1) - Number(b.round_no || 1) ||
        Number(a.seq_no || 0) - Number(b.seq_no || 0)
      )
      .map(x => ({
        seq_no: Number(x.seq_no),
        round_no: Number(x.round_no || 1),
        checkpoint_id: x.checkpoint_id,
        checkpoint_name: cpMap[x.checkpoint_id] ? cpMap[x.checkpoint_id].checkpoint_name : '',
        planned_time: x.planned_time,
        sla_min: Number(x.sla_min || 0)
      }));

    return {
      shift_id: s.shift_id,
      date: s.date,
      shift_name: s.shift_name,
      start_time: normalizeTimeValue(s.start_time),
      end_time: normalizeTimeValue(s.end_time),
      rounds_required: Number(s.rounds_required || 1),
      status: s.status,
      checkpoints: points
    };
  });
}

function submitCheckpoint(payload) {
  const required = ['shift_id', 'guard_id', 'qr_text_scanned', 'gps_lat', 'gps_lng', 'photo_url'];
  required.forEach(k => {
    if (payload[k] === undefined || payload[k] === null || payload[k] === '') {
      throw new Error(`Missing ${k}`);
    }
  });

  const now = new Date();
  const shiftId = payload.shift_id;
  const guardId = payload.guard_id;

  const checkpoints = getRows(SHEET.CHECKPOINTS).filter(c => String(c.active).toLowerCase() !== 'false');
  const cp = checkpoints.find(c => String(c.qr_text) === String(payload.qr_text_scanned));

  let status = 'INVALID_QR';
  let checkpointId = '';
  let distanceM = '';

  if (cp) {
    checkpointId = cp.checkpoint_id;
    distanceM = haversineMeters(Number(payload.gps_lat), Number(payload.gps_lng), Number(cp.lat), Number(cp.lng));
    if (distanceM > Number(cp.radius_m || 50)) {
      status = 'INVALID_GPS';
    } else {
      status = isLate(shiftId, checkpointId, now) ? 'LATE' : 'ONTIME';
    }
  }

  const photoResult = safeStorePhotoToDrive(payload.photo_url, {
    category: DRIVE.CHECKPOINT_SUBFOLDER,
    shift_id: shiftId,
    guard_id: guardId,
    checkpoint_id: checkpointId || 'unknown',
    prefix: 'checkpoint'
  });

  const row = {
    log_id: 'LOG-' + Utilities.getUuid(),
    shift_id: shiftId,
    guard_id: guardId,
    checkpoint_id: checkpointId,
    scan_time: fmtDateTime(now),
    qr_text_scanned: payload.qr_text_scanned,
    gps_lat: payload.gps_lat,
    gps_lng: payload.gps_lng,
    distance_m: distanceM === '' ? '' : Math.round(distanceM),
    photo_url: photoResult.url || '',
    status: status,
    remark: mergeRemark(payload.remark || '', photoResult.error)
  };

  appendRow(SHEET.CHECKLOGS, row);
  return row;
}

function submitIncident(payload) {
  const required = ['shift_id', 'guard_id', 'type', 'detail', 'photo_url', 'severity'];
  required.forEach(k => {
    if (!payload[k]) throw new Error(`Missing ${k}`);
  });

  const photoResult = safeStorePhotoToDrive(payload.photo_url, {
    category: DRIVE.INCIDENT_SUBFOLDER,
    shift_id: payload.shift_id,
    guard_id: payload.guard_id,
    checkpoint_id: payload.type || 'incident',
    prefix: 'incident'
  });

  const row = {
    incident_id: 'INC-' + Utilities.getUuid(),
    shift_id: payload.shift_id,
    guard_id: payload.guard_id,
    incident_time: fmtDateTime(new Date()),
    type: payload.type,
    detail: payload.detail,
    photo_url: photoResult.url || '',
    severity: payload.severity,
    status: 'OPEN'
  };
  if (photoResult.error) {
    row.detail = mergeRemark(row.detail, photoResult.error);
  }

  appendRow(SHEET.INCIDENTS, row);
  return row;
}

function closeShift(shiftId) {
  if (!shiftId) throw new Error('shiftId is required');

  const shifts = getRows(SHEET.SHIFTS);
  const shift = shifts.find(s => String(s.shift_id) === String(shiftId));
  if (!shift) throw new Error('Shift not found');

  const expectedRows = getRows(SHEET.SHIFT_CHECKPOINTS)
    .filter(x => String(x.shift_id) === String(shiftId));
  const expected = expectedRows.map(x => String(x.checkpoint_id));

  const validLogs = getRows(SHEET.CHECKLOGS).filter(l =>
    String(l.shift_id) === String(shiftId) &&
    (l.status === 'ONTIME' || l.status === 'LATE')
  );
  const effectiveGuardId = String(shift.guard_id || '').trim() || (validLogs[0] ? String(validLogs[0].guard_id || '') : '');

  const expectedCount = {};
  expected.forEach(cp => {
    expectedCount[cp] = Number(expectedCount[cp] || 0) + 1;
  });

  const checkedCount = {};
  const lateCount = {};
  validLogs.forEach(l => {
    const cp = String(l.checkpoint_id || '');
    if (!cp) return;
    checkedCount[cp] = Number(checkedCount[cp] || 0) + 1;
    if (l.status === 'LATE') lateCount[cp] = Number(lateCount[cp] || 0) + 1;
  });

  const totalPoints = expected.length;
  const checkedPoints = Object.keys(expectedCount).reduce((sum, cp) => {
    return sum + Math.min(Number(expectedCount[cp] || 0), Number(checkedCount[cp] || 0));
  }, 0);
  const latePoints = Object.keys(expectedCount).reduce((sum, cp) => {
    return sum + Math.min(Number(expectedCount[cp] || 0), Number(lateCount[cp] || 0));
  }, 0);
  const missedPoints = totalPoints - checkedPoints;
  const incidentsCount = getRows(SHEET.INCIDENTS).filter(i => String(i.shift_id) === String(shiftId)).length;
  const compliancePct = totalPoints > 0 ? Math.round((checkedPoints / totalPoints) * 10000) / 100 : 0;

  const summary = {
    shift_id: shift.shift_id,
    date: shift.date,
    guard_id: effectiveGuardId,
    supervisor_id: getSupervisorId(effectiveGuardId),
    total_points: totalPoints,
    checked_points: checkedPoints,
    late_points: latePoints,
    missed_points: missedPoints,
    incidents_count: incidentsCount,
    compliance_pct: compliancePct,
    closed_at: fmtDateTime(new Date())
  };

  upsertByKey(SHEET.SHIFT_SUMMARY, 'shift_id', summary);
  updateCellByKey(SHEET.SHIFTS, 'shift_id', shiftId, 'status', 'CLOSED');

  return summary;
}

/***** SUPERVISOR / ADMIN FUNCTIONS *****/
function supervisorLogin(supervisorId, password) {
  if (!supervisorId) throw new Error('supervisorId is required');

  const supervisors = getRows(SHEET.SUPERVISORS);
  const sup = supervisors.find(r =>
    String(r.supervisor_id) === String(supervisorId) &&
    String(r.status || '').toLowerCase() !== 'inactive'
  );

  if (!sup) throw new Error('Supervisor not found or inactive');
  const storedPassword = String(sup.password || '').trim();
  if (storedPassword && !verifyPassword(password, storedPassword)) {
    throw new Error('Invalid password');
  }

  return {
    supervisor_id: sup.supervisor_id,
    name: sup.name || '',
    email: sup.email || ''
  };
}

function getSupervisorDailySummary(supervisorId, date) {
  if (!supervisorId || !date) throw new Error('supervisorId and date are required');

  const shifts = getRows(SHEET.SHIFT_SUMMARY).filter(r =>
    String(r.supervisor_id) === String(supervisorId) &&
    String(r.date) === String(date)
  );

  const shiftIds = {};
  shifts.forEach(s => { shiftIds[String(s.shift_id)] = true; });

  const incidents = getRows(SHEET.INCIDENTS).filter(i => shiftIds[String(i.shift_id)]);

  const totalShifts = shifts.length;
  const totalChecked = shifts.reduce((a, b) => a + Number(b.checked_points || 0), 0);
  const totalLate = shifts.reduce((a, b) => a + Number(b.late_points || 0), 0);
  const totalMissed = shifts.reduce((a, b) => a + Number(b.missed_points || 0), 0);
  const totalIncidents = incidents.length;
  const avgCompliance = totalShifts
    ? (shifts.reduce((a, b) => a + Number(b.compliance_pct || 0), 0) / totalShifts)
    : 0;

  return {
    date: date,
    supervisor_id: supervisorId,
    kpi: {
      total_shifts: totalShifts,
      total_checked_points: totalChecked,
      total_late_points: totalLate,
      total_missed_points: totalMissed,
      total_incidents: totalIncidents,
      avg_compliance_pct: Math.round(avgCompliance * 100) / 100
    },
    shifts: shifts,
    incidents: incidents
  };
}

function getDashboardCharts(supervisorId, endDate, days) {
  if (!supervisorId) throw new Error('supervisorId is required');
  const end = endDate || fmtDateOnly(new Date());
  const dayCount = Math.max(1, Number(days || 30));
  const start = fmtDateOnly(addDays(parseDateOnly(end), -(dayCount - 1)));

  const guards = getRows(SHEET.GUARDS).filter(g => String(g.supervisor_id || '') === String(supervisorId));
  const guardSet = {};
  guards.forEach(g => { guardSet[String(g.guard_id)] = true; });

  const shifts = getRows(SHEET.SHIFTS).filter(s =>
    isDateInRange(s.date, start, end) &&
    (
      String(s.guard_id || '').trim() === '' ||
      String(s.guard_id || '').toUpperCase() === 'ANY' ||
      guardSet[String(s.guard_id)]
    )
  );
  const shiftIds = {};
  shifts.forEach(s => { shiftIds[String(s.shift_id)] = true; });

  const summaries = getRows(SHEET.SHIFT_SUMMARY).filter(r =>
    isDateInRange(r.date, start, end) &&
    String(r.supervisor_id || '') === String(supervisorId)
  );

  const checkLogs = getRows(SHEET.CHECKLOGS).filter(l => shiftIds[String(l.shift_id)]);
  const incidents = getRows(SHEET.INCIDENTS).filter(i => shiftIds[String(i.shift_id)]);
  const checkpointsMap = toMap(getRows(SHEET.CHECKPOINTS), 'checkpoint_id');
  const shiftCheckpointRows = getRows(SHEET.SHIFT_CHECKPOINTS).filter(r => shiftIds[String(r.shift_id)]);

  const dailyMap = {};
  buildDateRange(start, end).forEach(d => {
    dailyMap[d] = { date: d, compliance_pct: 0, checked: 0, missed: 0, invalid: 0, shift_count: 0 };
  });

  summaries.forEach(s => {
    const d = String(s.date);
    if (!dailyMap[d]) return;
    dailyMap[d].checked += Number(s.checked_points || 0);
    dailyMap[d].missed += Number(s.missed_points || 0);
    dailyMap[d].compliance_pct += Number(s.compliance_pct || 0);
    dailyMap[d].shift_count += 1;
  });

  checkLogs.forEach(l => {
    const d = toDateKey(l.scan_time);
    if (!dailyMap[d]) return;
    if (String(l.status || '').startsWith('INVALID')) {
      dailyMap[d].invalid += 1;
    }
  });

  const complianceTrend = Object.keys(dailyMap).sort().map(d => {
    const x = dailyMap[d];
    return {
      date: d,
      compliance_pct: x.shift_count ? Math.round((x.compliance_pct / x.shift_count) * 100) / 100 : 0
    };
  });

  const dailyOperations = Object.keys(dailyMap).sort().map(d => {
    const x = dailyMap[d];
    return { date: d, checked: x.checked, missed: x.missed, invalid: x.invalid };
  });

  const incidentsBySeverity = { LOW: 0, MEDIUM: 0, HIGH: 0 };
  incidents.forEach(i => {
    const key = String(i.severity || 'MEDIUM').toUpperCase();
    if (incidentsBySeverity[key] === undefined) incidentsBySeverity[key] = 0;
    incidentsBySeverity[key] += 1;
  });

  const summaryMap = toMap(summaries, 'shift_id');
  const incidentsByShift = {};
  incidents.forEach(i => {
    const sid = String(i.shift_id || '');
    incidentsByShift[sid] = Number(incidentsByShift[sid] || 0) + 1;
  });

  const shiftTypeAgg = {
    DAY: { shifts: 0, compliance_sum: 0, incidents: 0 },
    NIGHT: { shifts: 0, compliance_sum: 0, incidents: 0 }
  };
  shifts.forEach(s => {
    const sid = String(s.shift_id || '');
    const type = detectShiftType(s.shift_name);
    const sumRow = summaryMap[sid];
    shiftTypeAgg[type].shifts += 1;
    shiftTypeAgg[type].compliance_sum += Number(sumRow ? sumRow.compliance_pct : 0);
    shiftTypeAgg[type].incidents += Number(incidentsByShift[sid] || 0);
  });

  const shiftTypePerformance = Object.keys(shiftTypeAgg).map(type => {
    const x = shiftTypeAgg[type];
    return {
      shift_type: type,
      total_shifts: x.shifts,
      avg_compliance_pct: x.shifts ? Math.round((x.compliance_sum / x.shifts) * 100) / 100 : 0,
      incidents: x.incidents
    };
  });

  const expectedByShiftCp = {};
  shiftCheckpointRows.forEach(r => {
    const sid = String(r.shift_id || '');
    const cp = String(r.checkpoint_id || '');
    if (!sid || !cp) return;
    const key = `${sid}__${cp}`;
    expectedByShiftCp[key] = Number(expectedByShiftCp[key] || 0) + 1;
  });

  const validByShiftCp = {};
  const invalidByCp = {};
  checkLogs.forEach(l => {
    const sid = String(l.shift_id || '');
    const cp = String(l.checkpoint_id || '');
    const status = String(l.status || '');
    if (!cp) return;
    const key = `${sid}__${cp}`;
    if (status === 'ONTIME' || status === 'LATE') {
      validByShiftCp[key] = Number(validByShiftCp[key] || 0) + 1;
    } else if (status.startsWith('INVALID')) {
      invalidByCp[cp] = Number(invalidByCp[cp] || 0) + 1;
    }
  });

  const issueByCp = {};
  Object.keys(expectedByShiftCp).forEach(key => {
    const parts = key.split('__');
    const cp = parts[1];
    const expected = Number(expectedByShiftCp[key] || 0);
    const valid = Number(validByShiftCp[key] || 0);
    const missed = Math.max(0, expected - valid);
    issueByCp[cp] = Number(issueByCp[cp] || 0) + missed;
  });
  Object.keys(invalidByCp).forEach(cp => {
    issueByCp[cp] = Number(issueByCp[cp] || 0) + Number(invalidByCp[cp] || 0);
  });

  const topProblemCheckpoints = Object.keys(issueByCp)
    .map(cp => ({
      checkpoint_id: cp,
      checkpoint_name: checkpointsMap[cp] ? checkpointsMap[cp].checkpoint_name : '',
      issues: Number(issueByCp[cp] || 0)
    }))
    .filter(x => x.issues > 0)
    .sort((a, b) => b.issues - a.issues)
    .slice(0, 10);

  return {
    period: { start_date: start, end_date: end, days: dayCount },
    compliance_trend: complianceTrend,
    daily_operations: dailyOperations,
    incidents_by_severity: incidentsBySeverity,
    shift_type_performance: shiftTypePerformance,
    top_problem_checkpoints: topProblemCheckpoints
  };
}

function upsertSupervisor(payload) {
  if (!payload.supervisor_id) throw new Error('supervisor_id is required');
  if (!payload.name) throw new Error('name is required');
  const existing = getRows(SHEET.SUPERVISORS).find(r => String(r.supervisor_id) === String(payload.supervisor_id)) || null;

  const row = {
    supervisor_id: payload.supervisor_id,
    name: payload.name,
    email: payload.email || '',
    status: payload.status || 'active',
    password: payload.password ? hashPassword(payload.password) : (existing ? String(existing.password || '') : '')
  };

  upsertByKey(SHEET.SUPERVISORS, 'supervisor_id', row);
  return {
    supervisor_id: row.supervisor_id,
    name: row.name,
    email: row.email,
    status: row.status
  };
}

function changeSupervisorPassword(supervisorId, newPassword) {
  if (!supervisorId) throw new Error('supervisorId is required');
  if (!newPassword || String(newPassword).length < 6) {
    throw new Error('newPassword must be at least 6 characters');
  }

  const ok = updateCellByKey(SHEET.SUPERVISORS, 'supervisor_id', supervisorId, 'password', hashPassword(newPassword));
  if (!ok) {
    throw new Error('Cannot update password. Please add "password" column in Supervisors sheet.');
  }
  return { supervisor_id: supervisorId, changed: true };
}

function listGuards(supervisorId) {
  let rows = getRows(SHEET.GUARDS);
  if (supervisorId) {
    const filtered = rows.filter(r => String(r.supervisor_id || '') === String(supervisorId));
    // Fallback: if no guard is mapped to this supervisor, return all guards
    // so admin screens can still operate in setups without supervisor mapping.
    rows = filtered.length ? filtered : rows;
  }
  return rows;
}

function upsertGuard(payload) {
  if (!payload.guard_id) throw new Error('guard_id is required');
  if (!payload.name) throw new Error('name is required');

function generateShiftsFromTemplates(dateStr, supervisorId) {
  const date = dateStr || fmtDateOnly(new Date());
  const templates = getRows(SHEET.SHIFT_TEMPLATES).filter(t =>
    String(t.status || '').toUpperCase() !== 'INACTIVE'
  );
  const allowedGuardIds = {};
  if (supervisorId) {
    getRows(SHEET.GUARDS)
      .filter(g => String(g.supervisor_id || '') === String(supervisorId))
      .forEach(g => { allowedGuardIds[String(g.guard_id || '')] = true; });
  }

  let created = 0;
  templates.forEach(t => {
    const guardIds = normalizeGuardIds(t.guard_ids || t.guard_id || '');
    guardIds.forEach(gid => {
      if (supervisorId && !allowedGuardIds[String(gid)]) return;
      const shiftId = makeShiftIdFromTemplate(t.template_id, date, gid);
      const roundsRequired = Math.max(1, Number(t.rounds_per_shift || 1));
      const shift = {
        shift_id: shiftId,
        date: date,
        guard_id: gid,
        shift_name: t.shift_name || t.template_name,
        start_time: t.start_time || '08:00:00',
        end_time: t.end_time || '17:00:00',
        rounds_required: roundsRequired,
        template_id: t.template_id,
        status: 'OPEN'
      };

      upsertByKey(SHEET.SHIFTS, 'shift_id', shift);
      ensureShiftCheckpointsFromTemplate(shiftId, t.template_id, roundsRequired);
      created += 1;
    });
  });

  return { date: date, total: created };
}

function ensureGuardShiftsFromTemplates(guardId, dateStr) {
  const gid = String(guardId || '').trim();
  const date = String(dateStr || '').slice(0, 10);
  if (!gid || !date) return;

  const templates = getRows(SHEET.SHIFT_TEMPLATES).filter(t => {
    if (String(t.status || '').toUpperCase() === 'INACTIVE') return false;
    const guardIds = normalizeGuardIds(t.guard_ids || t.guard_id || '');
    return guardIds.indexOf(gid) >= 0;
  });

  templates.forEach(t => {
    const roundsRequired = Math.max(1, Number(t.rounds_per_shift || 1));
    const shiftId = makeShiftIdFromTemplate(t.template_id, date, gid);
    const shift = {
      shift_id: shiftId,
      date: date,
      guard_id: gid,
      shift_name: t.shift_name || t.template_name,
      start_time: t.start_time || '08:00:00',
      end_time: t.end_time || '17:00:00',
      rounds_required: roundsRequired,
      template_id: t.template_id,
      status: 'OPEN'
    };

    upsertByKey(SHEET.SHIFTS, 'shift_id', shift);
    ensureShiftCheckpointsFromTemplate(shiftId, t.template_id, roundsRequired);
  });
}

function ensureShiftCheckpointsFromTemplate(shiftId, templateId, roundsRequired) {
  const tRows = getRows(SHEET.TEMPLATE_CHECKPOINTS)
    .filter(r => String(r.template_id) === String(templateId))
    .sort((a, b) => Number(a.seq_no || 0) - Number(b.seq_no || 0));

  const expanded = [];
  for (let round = 1; round <= roundsRequired; round++) {
    tRows.forEach(r => {
      expanded.push({
        shift_id: shiftId,
        seq_no: Number(r.seq_no || 0),
        round_no: round,
        checkpoint_id: r.checkpoint_id,
        planned_time: '',
        sla_min: 0
      });
    });
  }

  const existing = getRows(SHEET.SHIFT_CHECKPOINTS)
    .filter(r => String(r.shift_id) === String(shiftId))
    .map(r => ({
      seq_no: Number(r.seq_no || 0),
      round_no: Number(r.round_no || 1),
      checkpoint_id: String(r.checkpoint_id || '')
    }))
    .sort((a, b) =>
      Number(a.round_no || 1) - Number(b.round_no || 1) ||
      Number(a.seq_no || 0) - Number(b.seq_no || 0)
    );

  const desired = expanded.map(r => ({
    seq_no: Number(r.seq_no || 0),
    round_no: Number(r.round_no || 1),
    checkpoint_id: String(r.checkpoint_id || '')
  }));

  const same = existing.length === desired.length &&
    existing.every((r, i) =>
      Number(r.round_no) === Number(desired[i].round_no) &&
      Number(r.seq_no) === Number(desired[i].seq_no) &&
      String(r.checkpoint_id) === String(desired[i].checkpoint_id)
    );

  if (same) return;

  deleteRowsByKey(SHEET.SHIFT_CHECKPOINTS, 'shift_id', shiftId);
  expanded.forEach(x => appendRow(SHEET.SHIFT_CHECKPOINTS, x));
}

function normalizeGuardIds(value) {
  if (Array.isArray(value)) {
    const uniq = {};
    const out = [];
    value.forEach(v => {
      const s = String(v || '').trim();
      if (!s) return;
      if (uniq[s]) return;
      uniq[s] = true;
      out.push(s);
    });
    return out;
  }
  return parseGuardIds(value);
}

function hashPassword(password) {
  const digest = Utilities.computeDigest(Utilities.DigestAlgorithm.SHA_256, String(password || ''), Utilities.Charset.UTF_8);
  const hex = digest.map(function(b) {
    const v = (b + 256) % 256;
    return ('0' + v.toString(16)).slice(-2);
  }).join('');
  return 'sha256:' + hex;
}

function verifyPassword(inputPassword, storedPassword) {
  const incoming = String(inputPassword || '');
  const stored = String(storedPassword || '').trim();
  if (!stored) return true;
  if (stored.indexOf('sha256:') === 0) {
    return stored === hashPassword(incoming);
  }
  return stored === incoming;
}

function storePhotoToDrive(photoValue, meta) {
  const source = String(photoValue || '').trim();
  if (!source) return '';
  if (!/^data:/i.test(source)) return source;

  const match = source.match(/^data:(.*?);base64,(.*)$/);
  if (!match) return source;

  const mimeType = String(match[1] || 'image/jpeg').trim() || 'image/jpeg';
  const base64Data = match[2] || '';
  const bytes = Utilities.base64Decode(base64Data);
  const extension = getFileExtensionFromMime(mimeType);
  const fileName = buildPhotoFileName(meta, extension);
  const folder = getPhotoTargetFolder(meta && meta.category ? meta.category : DRIVE.CHECKPOINT_SUBFOLDER);
  const blob = Utilities.newBlob(bytes, mimeType, fileName);
  const file = folder.createFile(blob);

  try {
    file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
  } catch (err) {
    // If sharing policy blocks public links, keep the native Drive URL instead.
  }

  return file.getUrl();
}

function safeStorePhotoToDrive(photoValue, meta) {
  try {
    return { url: storePhotoToDrive(photoValue, meta), error: '' };
  } catch (err) {
    // Fallback: do not fail guard checkpoint flow if Drive permission is missing.
    const source = String(photoValue || '').trim();
    if (/^https?:\/\//i.test(source)) return { url: source, error: '' };
    return {
      url: '',
      error: `DRIVE_UPLOAD_FAILED: ${String(err && err.message ? err.message : err)}`
    };
  }
}

function mergeRemark(baseText, extraText) {
  const a = String(baseText || '').trim();
  const b = String(extraText || '').trim();
  if (!a && !b) return '';
  if (!a) return b;
  if (!b) return a;
  return `${a} | ${b}`;
}

function getPhotoTargetFolder(subfolderName) {
  const scriptProps = PropertiesService.getScriptProperties();
  const configuredFolderId = String(
    scriptProps.getProperty(DRIVE.FOLDER_ID_PROPERTY) || DRIVE.DEFAULT_FOLDER_ID || ''
  ).trim();

  if (configuredFolderId) {
    try {
      const baseFolder = DriveApp.getFolderById(configuredFolderId);
      return getOrCreateSubfolder(baseFolder, subfolderName);
    } catch (err) {
      // Fallback to root folder by name when configured ID is invalid/unreachable.
    }
  }

  const rootFolder = getOrCreateRootFolder(DRIVE.ROOT_FOLDER_NAME);
  return getOrCreateSubfolder(rootFolder, subfolderName);
}

function getOrCreateRootFolder(folderName) {
  const folders = DriveApp.getFoldersByName(folderName);
  if (folders.hasNext()) return folders.next();
  return DriveApp.createFolder(folderName);
}

function getOrCreateSubfolder(parentFolder, folderName) {
  const folders = parentFolder.getFoldersByName(folderName);
  if (folders.hasNext()) return folders.next();
  return parentFolder.createFolder(folderName);
}

function getFileExtensionFromMime(mimeType) {
  const mime = String(mimeType || '').toLowerCase();
  if (mime.indexOf('png') >= 0) return 'png';
  if (mime.indexOf('webp') >= 0) return 'webp';
  if (mime.indexOf('gif') >= 0) return 'gif';
  return 'jpg';
}

function buildPhotoFileName(meta, extension) {
  const prefix = sanitizeFilePart(meta && meta.prefix ? meta.prefix : 'photo');
  const shiftId = sanitizeFilePart(meta && meta.shift_id ? meta.shift_id : 'shift');
  const guardId = sanitizeFilePart(meta && meta.guard_id ? meta.guard_id : 'guard');
  const pointId = sanitizeFilePart(meta && meta.checkpoint_id ? meta.checkpoint_id : 'point');
  const timestamp = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyyMMdd-HHmmss');
  return [prefix, shiftId, guardId, pointId, timestamp].join('_') + '.' + extension;
}

function sanitizeFilePart(value) {
  return String(value || '')
    .trim()
    .replace(/[^\w.-]+/g, '-')
    .replace(/^-+|-+$/g, '') || 'item';
}

function parseGuardIds(raw) {
  const txt = String(raw || '');
  const uniq = {};
  return txt
    .split(',')
    .map(x => String(x || '').trim())
    .filter(x => {
      if (!x) return false;
      if (uniq[x]) return false;
      uniq[x] = true;
      return true;
    });
}

function makeShiftIdFromTemplate(templateId, dateStr, guardId) {
  const d = String(dateStr || '').replace(/-/g, '');
  const g = String(guardId || '').trim();
  if (g) return `${String(templateId)}-${g}-${d}`;
  return `${String(templateId)}-${d}`;
}

    group[sid].push(s);
  });

  Object.keys(group).forEach(sid => {
    const rows = group[sid];
    const sup = supervisors[sid];
    const toEmail = sup && sup.email ? sup.email : '';

    const totalShifts = rows.length;
    const totalMissed = sum(rows, 'missed_points');
    const totalLate = sum(rows, 'late_points');
    const totalIncidents = sum(rows, 'incidents_count');
    const avgCompliance = totalShifts ? (sum(rows, 'compliance_pct') / totalShifts).toFixed(2) : '0.00';

    const subject = `Guard Tour Daily Summary (${targetDate})`;
    const html = `
      <h3>Guard Tour Daily Summary: ${targetDate}</h3>
      <p>Total shifts: <b>${totalShifts}</b></p>
      <p>Missed points: <b>${totalMissed}</b></p>
      <p>Late points: <b>${totalLate}</b></p>
      <p>Total incidents: <b>${totalIncidents}</b></p>
      <p>Average compliance: <b>${avgCompliance}%</b></p>
    `;

    try {
      if (toEmail) {
        MailApp.sendEmail({ to: toEmail, subject: subject, htmlBody: html });
        appendRow(SHEET.DAILY_EMAIL_LOG, {
          log_date: targetDate,
          send_time: fmtDateTime(new Date()),
          recipient_email: toEmail,
          total_shifts: totalShifts,
          total_missed: totalMissed,
          total_late: totalLate,
          total_incidents: totalIncidents,
          status: 'SENT',
          error_message: ''
        });
      }
    } catch (err) {
      appendRow(SHEET.DAILY_EMAIL_LOG, {
        log_date: targetDate,
        send_time: fmtDateTime(new Date()),
        recipient_email: toEmail,
        total_shifts: totalShifts,
        total_missed: totalMissed,
        total_late: totalLate,
        total_incidents: totalIncidents,
        status: 'FAILED',
        error_message: String(err.message || err)
      });
    }
  });
}

function isLate(shiftId, checkpointId, scanDate) {
  const sc = getRows(SHEET.SHIFT_CHECKPOINTS).find(x =>
    String(x.shift_id) === String(shiftId) && String(x.checkpoint_id) === String(checkpointId)
  );
  if (!sc || !sc.planned_time) return false;

  const shifts = getRows(SHEET.SHIFTS);
  const shift = shifts.find(s => String(s.shift_id) === String(shiftId));
  if (!shift) return false;

  const planned = parseDateTimeFromShiftDate(shift.date, sc.planned_time);
  if (!planned) return false;
  planned.setMinutes(planned.getMinutes() + Number(sc.sla_min || 0));

  return scanDate.getTime() > planned.getTime();
}

function getSupervisorId(guardId) {
  const g = getRows(SHEET.GUARDS).find(x => String(x.guard_id) === String(guardId));
  return g ? g.supervisor_id : '';
}

function getRows(sheetName) {
  const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName);
  if (!sh) throw new Error(`Sheet not found: ${sheetName}`);
  const values = sh.getDataRange().getValues();
  if (values.length < 2) return [];
  const headers = values[0].map(String);
  return values.slice(1).map(r => {
    const obj = {};
    headers.forEach((h, i) => obj[h] = r[i]);
    return obj;
  });
}

function appendRow(sheetName, obj) {
  const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName);
  if (!sh) throw new Error(`Sheet not found: ${sheetName}`);
  const headers = sh.getRange(1, 1, 1, sh.getLastColumn()).getValues()[0].map(String);
  const row = headers.map(h => obj[h] !== undefined ? obj[h] : '');
  sh.appendRow(row);
}

function upsertByKey(sheetName, keyField, obj) {
  const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName);
  const values = sh.getDataRange().getValues();
  const headers = values[0].map(String);
  const keyIdx = headers.indexOf(keyField);
  if (keyIdx < 0) throw new Error(`Key field not found: ${keyField}`);

  const rowIdx = values.findIndex((r, i) => i > 0 && String(r[keyIdx]) === String(obj[keyField]));
  const row = headers.map(h => obj[h] !== undefined ? obj[h] : '');

  if (rowIdx > 0) sh.getRange(rowIdx + 1, 1, 1, headers.length).setValues([row]);
  else sh.appendRow(row);
}

function updateCellByKey(sheetName, keyField, keyValue, targetField, targetValue) {
  const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName);
  const values = sh.getDataRange().getValues();
  const headers = values[0].map(String);
  const keyIdx = headers.indexOf(keyField);
  const targetIdx = headers.indexOf(targetField);
  if (keyIdx < 0 || targetIdx < 0) return false;

  for (let i = 1; i < values.length; i++) {
    if (String(values[i][keyIdx]) === String(keyValue)) {
      sh.getRange(i + 1, targetIdx + 1).setValue(targetValue);
      return true;
    }
  }
  return false;
}

function deleteRowsByKey(sheetName, keyField, keyValue) {
  const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName);
  if (!sh) throw new Error(`Sheet not found: ${sheetName}`);

  const values = sh.getDataRange().getValues();
  if (values.length < 2) return;

  const headers = values[0].map(String);
  const keyIdx = headers.indexOf(keyField);
  if (keyIdx < 0) throw new Error(`Key field not found: ${keyField}`);
  let deleted = 0;

  for (let i = values.length - 1; i >= 1; i--) {
    if (String(values[i][keyIdx]) === String(keyValue)) {
      sh.deleteRow(i + 1);
      deleted += 1;
    }
  }
  return deleted;
}


function normalizeTimeValue(value) {
  if (value === null || value === undefined || value === '') return '';

  if (Object.prototype.toString.call(value) === '[object Date]' && !isNaN(value.getTime())) {
    return Utilities.formatDate(value, Session.getScriptTimeZone(), 'HH:mm:ss');
  }

  const s = String(value).trim();
  if (!s) return '';

  const hm = s.match(/^(\d{1,2}):(\d{2})(?::(\d{2}))?$/);
  if (hm) {
    const hh = String(Math.min(23, Math.max(0, Number(hm[1])))).padStart(2, '0');
    const mm = String(Math.min(59, Math.max(0, Number(hm[2])))).padStart(2, '0');
    const ss = String(Math.min(59, Math.max(0, Number(hm[3] || 0)))).padStart(2, '0');
    return `${hh}:${mm}:${ss}`;
  }

  const d = new Date(s);
  if (!isNaN(d.getTime())) {
    return Utilities.formatDate(d, Session.getScriptTimeZone(), 'HH:mm:ss');
  }

  return s;
}

function parseDateOnly(dateVal) {
  if (Object.prototype.toString.call(dateVal) === '[object Date]' && !isNaN(dateVal.getTime())) {
    const d = new Date(dateVal);
    d.setHours(0, 0, 0, 0);
    return d;
  }
  const s = String(dateVal || '').slice(0, 10);
  const d = new Date(`${s}T00:00:00`);
  if (isNaN(d.getTime())) throw new Error(`Invalid date: ${dateVal}`);
  return d;
}

function isDateInRange(dateVal, startYmd, endYmd) {
  const d = String(dateVal || '').slice(0, 10);
  return d >= String(startYmd) && d <= String(endYmd);
}

function buildDateRange(startYmd, endYmd) {
  const out = [];
  let cur = parseDateOnly(startYmd);
  const end = parseDateOnly(endYmd);
  while (cur.getTime() <= end.getTime()) {
    out.push(fmtDateOnly(cur));
    cur = addDays(cur, 1);
  }
  return out;
}

function detectShiftType(shiftName) {
  const s = String(shiftName || '').toLowerCase();
  if (
    s.indexOf('night') >= 0 ||
    s.indexOf('กลางคืน') >= 0 ||
    s.indexOf('ดึก') >= 0
  ) {
    return 'NIGHT';
  }
  return 'DAY';
}

function toMap(arr, key) {
  const out = {};
  arr.forEach(x => out[String(x[key])] = x);
  return out;
}

function sum(arr, key) {
  return arr.reduce((a, b) => a + Number(b[key] || 0), 0);
}

function fmtDateOnly(d) {

}

function toDateKey(value) {
  if (value === null || value === undefined || value === '') return '';
  if (Object.prototype.toString.call(value) === '[object Date]' && !isNaN(value.getTime())) {
    return Utilities.formatDate(value, Session.getScriptTimeZone(), 'yyyy-MM-dd');
  }
  const s = String(value).trim();
  if (!s) return '';
  if (/^\d{4}-\d{2}-\d{2}$/.test(s)) return s;
  const parsed = new Date(s);
  if (!isNaN(parsed.getTime())) {
    return Utilities.formatDate(parsed, Session.getScriptTimeZone(), 'yyyy-MM-dd');
  }
  return s.slice(0, 10);
}

function fmtDateTime(d) {
  return Utilities.formatDate(d, Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm:ss');
}

function normalizeDateTimeValue(value) {
  if (value === null || value === undefined || value === '') return '';
  if (Object.prototype.toString.call(value) === '[object Date]' && !isNaN(value.getTime())) {
    return Utilities.formatDate(value, Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm:ss');
  }
  const s = String(value).trim();
  if (!s) return '';
  const d = new Date(s);
  if (!isNaN(d.getTime())) {
    return Utilities.formatDate(d, Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm:ss');
  }

}

function toDateTimeMillis(value) {
  if (value === null || value === undefined || value === '') return 0;
  if (Object.prototype.toString.call(value) === '[object Date]' && !isNaN(value.getTime())) {
    return value.getTime();
  }
  const d = new Date(String(value));
  if (!isNaN(d.getTime())) return d.getTime();
  return 0;
}

function addDays(d, n) {
  const x = new Date(d);
  x.setDate(x.getDate() + n);
  return x;
}

function parseDateTimeFromShiftDate(dateVal, timeVal) {
  let dateStr = '';
  if (Object.prototype.toString.call(dateVal) === '[object Date]' && !isNaN(dateVal.getTime())) {
    dateStr = Utilities.formatDate(dateVal, Session.getScriptTimeZone(), 'yyyy-MM-dd');
  } else {
    dateStr = String(dateVal).slice(0, 10);
  }

  const t = String(timeVal).trim();
  if (!dateStr || !t) return null;
  const iso = `${dateStr}T${t.length === 5 ? t + ':00' : t}`;
  const d = new Date(iso);
  return isNaN(d.getTime()) ? null : d;
}

function haversineMeters(lat1, lon1, lat2, lon2) {
  const R = 6371000;
  const toRad = x => x * Math.PI / 180;
  const dLat = toRad(lat2 - lat1);
  const dLon = toRad(lon2 - lon1);
  const a =
    Math.sin(dLat / 2) * Math.sin(dLat / 2) +
    Math.cos(toRad(lat1)) * Math.cos(toRad(lat2)) *
    Math.sin(dLon / 2) * Math.sin(dLon / 2);
  return 2 * R * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
}
